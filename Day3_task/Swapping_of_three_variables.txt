# To swap three variables 

a = int(input("Enter first integer: "))
b = int(input("Enter second integer: "))
print("Befoer swapping")
print("The value of first integer is: ", a)
print("The value of second integer is: ", b)
c = a
a = b
b = c
print("After swapping")
print("The value of first integer is: ", a)
print("The value of second integer is: ", b)
